from pyrogram import Client, filters
from pyrogram.types import Message
import os
import json
from dotenv import load_dotenv

load_dotenv()

API_ID = int(os.getenv("API_ID"))
API_HASH = os.getenv("API_HASH")
BOT_TOKEN = os.getenv("BOT_TOKEN")
STORAGE_CHAT_ID = int(os.getenv("STORAGE_CHAT_ID"))

# Load file database
if os.path.exists("files.json"):
    with open("files.json") as f:
        FILE_DB = json.load(f)
else:
    FILE_DB = {}

def save_db():
    with open("files.json", "w") as f:
        json.dump(FILE_DB, f)

app = Client("filestreambot", api_id=API_ID, api_hash=API_HASH, bot_token=BOT_TOKEN)
@app.on_message(filters.private & filters.command("help"))
async def help_command(client, message: Message):
    await message.reply_text(
        "ğŸ‘‹ **Welcome to the File Stream Bot!**\n\n"
        "Just send me any file (document, video, audio, photo) and I'll give you a shareable download link.\n\n"
        "**Commands:**\n"
        "/start - Get a welcome message or retrieve a file using a share link.\n"
        "/help - Show this help message."
    )

@app.on_message(filters.private & (filters.document | filters.video | filters.audio | filters.photo))
async def save_file(client, message: Message):
    sent = await message.forward(STORAGE_CHAT_ID)
    file_id = str(sent.id)

    media = message.document or message.video or message.audio or message.photo
    file_name = getattr(media, "file_name", "Unknown")
    file_size = getattr(media, "file_size", 0)
    file_type = message.media
    file_tg_id = media.file_id

    FILE_DB[file_id] = {
        "file_type": file_type,
        "file_id": file_tg_id,
        "file_name": file_name,
        "file_size": file_size
    }
    save_db()

    start_link = f"https://t.me/{(await client.get_me()).username}?start={file_id}"

    await message.reply_text(
        f"ğŸ“¤ **File Uploaded!**\n\n"
        f"ğŸ“ Name: `{file_name}`\n"
        f"ğŸ“ Size: `{round(file_size / 1024 / 1024, 2)} MB`\n"
        f"ğŸ“¦ Type: `{file_type}`\n"
        f"âš™ï¸ Hash: `{file_id}`\n\n"
        f"ğŸ”— **Share Link:**\n[Click Here]({start_link})",
        disable_web_page_preview=True
    )

@app.on_message(filters.private & filters.command("start"))
async def send_file(client, message: Message):
    args = message.text.split(" ")
    if len(args) == 2:
        file_id = args[1]
        if file_id in FILE_DB:
            await message.reply_document(
                FILE_DB[file_id]["file_id"],
                caption=f"ğŸ“¥ **Here's your file!**\nğŸ“ {FILE_DB[file_id]['file_name']}"
            )
        else:
            await message.reply("âŒ File not found or expired.")
    else:
        await message.reply("ğŸ‘‹ Send me a file and Iâ€™ll give you a shareable download link!")

app.run()
